ENUMERATION ALGORITHM
def enumeration ()

    maximum sum = 0
    start = 0
    stop = 0
    
    for loop through array(start, end of array)
        current start = x
        
        for loop through array(x, end of array)
            current sum = 0
            current stop = x
            
            for loop through array (x, x)
                current sum + array[x]
                if current sum > maximum sum
                    maximum sum = current sum
                    start = current start
                    stop = current stop
                    
    return maximum sum
